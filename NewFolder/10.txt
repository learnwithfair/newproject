clc;
clear all;
close all;

n = -5:1:5;
y = ((n == 0) ==1);
subplot(3,1,1);
stem(n,y);
xlabel("--------------> n");
ylabel('Amplitude');
title('Unit Sample Signal');
grid on;

x = (n>=0)==1;
subplot(3,1,2);
stem(n,x);
xlabel("--------------> n");
ylabel('Amplitude');
title('Unit Step Signal');
grid on;

k = (((n>=1)==1).*n);
subplot(3,1,3);
stem(n,k);
xlabel("--------------> n");
ylabel('Amplitude');
title('Unit Ramp Signal');
grid on;
