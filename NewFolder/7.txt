clc;
clear all;
close all;

f = -2:0.1:2;
x = 4*sinc(4*f);
subplot(3,1,1)
plot(f,real(x));
xlabel('f------------->');
ylabel('Amplitude');
title('Real Part');
grid on;


subplot(3,1,2);
plot(f,abs(x));
xlabel('f------------->');
ylabel('Amplitude');
title('Magnitude Part');
grid on;

subplot(3,1,3);
plot(f,angle(x));
title("Phase Part");
xlabel('f------------->');
ylabel('Amplitude');
grid on;
